server {
    server_name backend.documentreader.silverlinepos.com;

    client_max_body_size 50M;   # allow up to 50 MB uploads

    location / {
        proxy_pass http://127.0.0.1:8015;   # must match Gunicorn port
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Fix timeouts to match Gunicorn (300s)
        proxy_connect_timeout   300s;
        proxy_send_timeout      300s;
        proxy_read_timeout      300s;
        send_timeout            300s;

        # Allow iframe embedding
        add_header X-Frame-Options "ALLOWALL";
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/backend.documentreader.silverlinepos.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/backend.documentreader.silverlinepos.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}

server {
    if ($host = backend.documentreader.silverlinepos.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    listen 80;
    server_name backend.documentreader.silverlinepos.com;
    return 404; # managed by Certbot
}
